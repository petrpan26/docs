## Welcome to Viggle API

Generate AI-powered character animation videos. Create reusable character and scene assets, then render videos with two modes: progressive chunk delivery for real-time playback, or batch processing for complete video download.

<CardGroup cols={2}>
  <Card title="Live Render" icon="bolt">
    Stream video chunks as they're generated. Ideal for real-time preview and progressive playback.
  </Card>
  <Card title="Batch Render" icon="layer-group">
    Wait for complete video. Simple workflow for background processing and file downloads.
  </Card>
</CardGroup>

## Base URL

All API requests should be made to:

```
https://api.viggle.ai
```

## Quick Start

Generate your first AI animation video in 5 steps:

1. **Get API Key** - Create one in your [Dashboard](https://viggle.ai/dashboard)
2. **Upload Character** - A reference image (PNG/JPG)
3. **Upload Scene** - A driving video (MP4/MOV)
4. **Render** - Combine character + scene
5. **Download** - Get your video

### Complete Example

Copy and run this script. Replace `YOUR_API_KEY` and file paths.

<CodeGroup>

```python Python
import requests
import time

API_KEY = "YOUR_API_KEY"
BASE = "https://api.viggle.ai"
headers = {"Authorization": f"Bearer {API_KEY}"}

def wait_for_ready(url):
    """Poll until status is ready or failed"""
    while True:
        resp = requests.get(url, headers=headers).json()
        if resp["status"] == "ready":
            return resp
        if resp["status"] == "failed":
            raise Exception(resp.get("error", "Processing failed"))
        time.sleep(1)

def wait_for_complete(url):
    """Poll until render is complete or failed"""
    while True:
        resp = requests.get(url, headers=headers).json()
        print(f"  Status: {resp['status']}")
        if resp["status"] == "complete":
            return resp
        if resp["status"] == "failed":
            raise Exception(resp.get("error", "Render failed"))
        time.sleep(1)

# Step 1: Create Character
print("1. Creating character...")
with open("character.png", "rb") as f:
    resp = requests.post(f"{BASE}/api/characters/preprocess",
        headers=headers, files={"image": f}, data={"name": "My Character"})
character_id = resp.json()["id"]
print(f"   Waiting for processing...")
wait_for_ready(f"{BASE}/api/characters/{character_id}")
print(f"   Done! ID: {character_id}")

# Step 2: Create Scene
print("2. Creating scene...")
with open("dance.mp4", "rb") as f:
    resp = requests.post(f"{BASE}/api/scenes/preprocess",
        headers=headers, files={"video": f}, data={"name": "Dance"})
scene_id = resp.json()["id"]
print(f"   Waiting for processing...")
wait_for_ready(f"{BASE}/api/scenes/{scene_id}")
print(f"   Done! ID: {scene_id}")

# Step 3: Render Video
print("3. Starting render...")
resp = requests.post(f"{BASE}/api/render", headers=headers,
    json={"character_id": character_id, "scene_id": scene_id})
job_id = resp.json()["job_id"]
print(f"   Job ID: {job_id}")
wait_for_complete(f"{BASE}/api/render/{job_id}")

# Step 4: Download
print("4. Downloading video...")
video = requests.get(f"{BASE}/api/render/{job_id}/video", headers=headers)
with open("output.mp4", "wb") as f:
    f.write(video.content)
print("Done! Saved to output.mp4")
```


```javascript JavaScript
const fs = require('fs');
const FormData = require('form-data');

const API_KEY = "YOUR_API_KEY";
const BASE = "https://api.viggle.ai";
const headers = { "Authorization": `Bearer ${API_KEY}` };

const sleep = (ms) => new Promise(r => setTimeout(r, ms));

async function waitForReady(url) {
  while (true) {
    const resp = await fetch(url, { headers }).then(r => r.json());
    if (resp.status === "ready") return resp;
    if (resp.status === "failed") throw new Error(resp.error || "Processing failed");
    await sleep(1000);
  }
}

async function waitForComplete(url) {
  while (true) {
    const resp = await fetch(url, { headers }).then(r => r.json());
    console.log(`  Status: ${resp.status}`);
    if (resp.status === "complete") return resp;
    if (resp.status === "failed") throw new Error(resp.error || "Render failed");
    await sleep(1000);
  }
}

async function main() {
  // Step 1: Create Character
  console.log("1. Creating character...");
  let form = new FormData();
  form.append("name", "My Character");
  form.append("image", fs.createReadStream("character.png"));
  let resp = await fetch(`${BASE}/api/characters/preprocess`, {
    method: "POST", headers: { ...headers, ...form.getHeaders() }, body: form
  }).then(r => r.json());
  const characterId = resp.id;
  console.log("   Waiting for processing...");
  await waitForReady(`${BASE}/api/characters/${characterId}`);
  console.log(`   Done! ID: ${characterId}`);

  // Step 2: Create Scene
  console.log("2. Creating scene...");
  form = new FormData();
  form.append("name", "Dance");
  form.append("video", fs.createReadStream("dance.mp4"));
  resp = await fetch(`${BASE}/api/scenes/preprocess`, {
    method: "POST", headers: { ...headers, ...form.getHeaders() }, body: form
  }).then(r => r.json());
  const sceneId = resp.id;
  console.log("   Waiting for processing...");
  await waitForReady(`${BASE}/api/scenes/${sceneId}`);
  console.log(`   Done! ID: ${sceneId}`);

  // Step 3: Render Video
  console.log("3. Starting render...");
  resp = await fetch(`${BASE}/api/render`, {
    method: "POST",
    headers: { ...headers, "Content-Type": "application/json" },
    body: JSON.stringify({ character_id: characterId, scene_id: sceneId })
  }).then(r => r.json());
  const jobId = resp.job_id;
  console.log(`   Job ID: ${jobId}`);
  await waitForComplete(`${BASE}/api/render/${jobId}`);

  // Step 4: Download
  console.log("4. Downloading video...");
  const video = await fetch(`${BASE}/api/render/${jobId}/video`, { headers });
  fs.writeFileSync("output.mp4", Buffer.from(await video.arrayBuffer()));
  console.log("Done! Saved to output.mp4");
}

main().catch(console.error);
```

## Rendering Modes

### Live Render (Progressive)

Use the **Poll endpoint** with long-polling to receive video chunks as they're rendered. Best for:

- Real-time preview
- Progressive playback
- Streaming applications

### Batch Render (Complete)

Use the **Status endpoint** to check progress and download the complete video when ready. Best for:

- Background processing
- File downloads
- Simple integration
