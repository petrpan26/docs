title: 'Live Rendering'
description: 'Stream video chunks in real-time using long-polling'
---

Live rendering lets you receive and play video chunks as they're generated, instead of waiting for the full video to complete. This is ideal for real-time previews and streaming applications.

## How It Works

Instead of polling the status endpoint and downloading at the end, you use the **Poll endpoint** which blocks until new chunks are ready:

```mermaid
sequenceDiagram
    participant Client
    participant API
    participant Server

    Client->>API: POST /api/render
    API-->>Client: { job_id, status: "queued" }

    loop Until complete
        Client->>API: GET /poll?last_chunk=-1
        Note over API: Blocks until chunk ready
        Server-->>API: Chunk 0 rendered
        API-->>Client: { new_chunks: [{chunk_id: 0, cdn_url: "..."}] }
        Client->>Client: Play chunk 0

        Client->>API: GET /poll?last_chunk=0
        Server-->>API: Chunk 1 rendered
        API-->>Client: { new_chunks: [{chunk_id: 1, cdn_url: "..."}] }
        Client->>Client: Play chunk 1
    end

    API-->>Client: { status: "complete", cdn_url: "..." }
```

## Basic Example

<CodeGroup>

```python Python
import requests

API_KEY = "YOUR_API_KEY"
BASE = "https://apis.viggle.ai"
headers = {"Authorization": f"Bearer {API_KEY}"}

# 1. Create render job
job = requests.post(f"{BASE}/api/render", headers=headers, data={
    "character_id": "char_xxx",
    "scene_id": "scene_xxx",
}).json()
print(f"Job: {job['job_id']}")

# 2. Long-poll for chunks
last_chunk = -1
all_chunks = []

while True:
    resp = requests.get(
        f"{BASE}/api/render/{job['job_id']}/poll",
        headers=headers,
        params={"last_chunk": last_chunk, "timeout": 30}
    ).json()

    # Process new chunks as they arrive
    for chunk in resp["new_chunks"]:
        print(f"  Chunk {chunk['chunk_id']} ready: {chunk['cdn_url']}")
        all_chunks.append(chunk)
        last_chunk = max(last_chunk, chunk["chunk_id"])

    if resp["status"] == "complete":
        print(f"Complete! Full video: {resp['cdn_url']}")
        break
    elif resp["status"] == "failed":
        raise Exception(resp.get("error"))
```

## Progressive Playback with fMP4 + MSE

Each chunk is a **fragmented MP4** (fMP4) file with `empty_moov` + `moof`/`mdat` fragment pairs. This means chunks can be fed directly into the browser's [Media Source Extensions (MSE)](https://developer.mozilla.org/en-US/docs/Web/API/Media_Source_Extensions_API) API for seamless, gapless playback as they arrive.

<Info>
  See [fMP4 Format Specification](/guides/fmp4-format) for details on the chunk file structure.
</Info>

### Poll-Based MSE Player

```javascript
const video = document.getElementById("player");
const mime = 'video/mp4; codecs="avc1.4d0028"';

async function streamWithMSE(jobId) {
  const headers = { Authorization: `Bearer ${API_KEY}` };
  const mediaSource = new MediaSource();
  video.src = URL.createObjectURL(mediaSource);

  await new Promise(r => mediaSource.addEventListener("sourceopen", r, { once: true }));
  const sourceBuffer = mediaSource.addSourceBuffer(mime);

  let lastChunk = -1;
  const queue = [];
  let appending = false;

  function drainQueue() {
    if (appending || queue.length === 0) return;
    appending = true;
    sourceBuffer.appendBuffer(queue.shift());
  }

  sourceBuffer.addEventListener("updateend", () => {
    appending = false;
    drainQueue();
  });

  while (true) {
    const resp = await fetch(
      `${BASE}/api/render/${jobId}/poll?last_chunk=${lastChunk}&timeout=30`,
      { headers }
    ).then(r => r.json());

    for (const chunk of resp.new_chunks) {
      const buf = await fetch(chunk.cdn_url).then(r => r.arrayBuffer());
      queue.push(buf);
      drainQueue();
      lastChunk = Math.max(lastChunk, chunk.chunk_id);
    }

    // Auto-play once first chunk is buffered
    if (video.paused && sourceBuffer.buffered.length > 0) {
      video.play();
    }

    if (resp.status === "complete") {
      // Wait for queue to drain, then signal end of stream
      const waitDrain = () => new Promise(r => {
        const check = () => (queue.length === 0 && !appending) ? r() : setTimeout(check, 50);
        check();
      });
      await waitDrain();
      mediaSource.endOfStream();
      break;
    }

    if (resp.status === "failed") {
      console.error("Render failed:", resp.error);
      break;
    }
  }
}
```

### Server-Side: Stream to File

Download fMP4 chunks as they arrive and concatenate into a single file:

<CodeGroup>

```python Python
import requests

API_KEY = "YOUR_API_KEY"
BASE = "https://apis.viggle.ai"
headers = {"Authorization": f"Bearer {API_KEY}"}

def stream_to_file(job_id, output_path="stream.mp4"):
    last_chunk = -1

    with open(output_path, "wb") as f:
        while True:
            resp = requests.get(
                f"{BASE}/api/render/{job_id}/poll",
                headers=headers,
                params={"last_chunk": last_chunk, "timeout": 30},
                timeout=45,
            ).json()

            for chunk in resp["new_chunks"]:
                data = requests.get(chunk["cdn_url"]).content
                f.write(data)
                f.flush()
                last_chunk = max(last_chunk, chunk["chunk_id"])
                print(f"  Wrote chunk {chunk['chunk_id']} ({len(data)} bytes)")

            if resp["status"] == "complete":
                print(f"Done! Saved to {output_path}")
                break
            elif resp["status"] == "failed":
                raise Exception(resp.get("error"))
```

```javascript JavaScript
const fs = require("fs");

const API_KEY = "YOUR_API_KEY";
const BASE = "https://apis.viggle.ai";
const headers = { Authorization: `Bearer ${API_KEY}` };

async function streamToFile(jobId, outputPath = "stream.mp4") {
  const writeStream = fs.createWriteStream(outputPath);
  let lastChunk = -1;

  while (true) {
    const resp = await fetch(
      `${BASE}/api/render/${jobId}/poll?last_chunk=${lastChunk}&timeout=30`,
      { headers, signal: AbortSignal.timeout(45000) }
    ).then(r => r.json());

    for (const chunk of resp.new_chunks) {
      const data = await fetch(chunk.cdn_url).then(r => r.arrayBuffer());
      writeStream.write(Buffer.from(data));
      lastChunk = Math.max(lastChunk, chunk.chunk_id);
      console.log(`  Wrote chunk ${chunk.chunk_id} (${data.byteLength} bytes)`);
    }

    if (resp.status === "complete") {
      writeStream.end();
      console.log(`Done! Saved to ${outputPath}`);
      break;
    }
    if (resp.status === "failed") {
      writeStream.end();
      throw new Error(resp.error);
    }
  }
}
```

```bash cURL
#!/bin/bash
# Stream fMP4 chunks to a single file as they arrive
API_KEY="YOUR_API_KEY"
BASE="https://apis.viggle.ai"
JOB_ID="YOUR_JOB_ID"
OUTPUT="stream.mp4"
LAST_CHUNK=-1

> "$OUTPUT"  # Create/truncate output file

while true; do
  RESP=$(curl -s "$BASE/api/render/$JOB_ID/poll?last_chunk=$LAST_CHUNK&timeout=30" \
    -H "Authorization: Bearer $API_KEY" --max-time 45)

  # Parse each chunk URL and download it (requires jq)
  echo "$RESP" | jq -r '.new_chunks[].cdn_url' | while read -r URL; do
    curl -s "$URL" >> "$OUTPUT"
  done

  # Update last_chunk to highest chunk_id received
  NEW_LAST=$(echo "$RESP" | jq '[.new_chunks[].chunk_id] | max // empty')
  [ -n "$NEW_LAST" ] && LAST_CHUNK=$NEW_LAST

  STATUS=$(echo "$RESP" | jq -r '.status')
  [ "$STATUS" = "complete" ] && echo "Done! Saved to $OUTPUT" && break
  [ "$STATUS" = "failed" ] && echo "Error: $(echo "$RESP" | jq -r '.error')" && exit 1
done
```

## fMP4 Chunk Format

Each chunk is a **fragmented MP4** (ISO BMFF) segment. Chunk 0 contains the initialization segment (`ftyp` + `moov`), subsequent chunks contain only media segments (`moof` + `mdat`).

Key properties:
- **Codec**: H.264 Main Profile, 25fps, no B-frames
- **PTS == DTS**: No composition time offsets (smooth byte-concatenation)
- **MSE codec string**: `'video/mp4; codecs="avc1.4d0028"'`

<Info>
  For the full specification including box layout, key fields, and byte-concatenation details, see [fMP4 Format Specification](/guides/fmp4-format).
</Info>

## Poll vs Status: When to Use Which

| Feature | Poll (`/poll`) | Status (`/`) |
|---------|----------------|--------------|
| **Auth required** | Yes | Yes |
| **Blocking** | Yes (long-poll up to 60s) | No (returns immediately) |
| **Returns chunks** | New chunks since `last_chunk` | All completed chunks |
| **Best for** | Real-time streaming, progressive playback | Background polling, simple integration |
| **Latency** | Low (returns as soon as chunk is ready) | 5s polling interval |

## Tips

<AccordionGroup>
  <Accordion title="Set appropriate timeouts">
    The `timeout` parameter (default 30s, max 60s) controls how long the server holds your request. Set your HTTP client timeout higher than this value to avoid premature disconnections.

    <CodeGroup>

    ```python Python
    # Set requests timeout higher than poll timeout
    resp = requests.get(url, params={"timeout": 30}, timeout=45)
    ```

    ```javascript JavaScript
    // Use AbortSignal.timeout to set a client-side timeout higher than poll timeout
    const resp = await fetch(url + "?timeout=30", {
      headers, signal: AbortSignal.timeout(45000)
    });
    ```

    ```bash cURL
    # Set --max-time higher than the poll timeout parameter
    curl "https://apis.viggle.ai/api/render/JOB_ID/poll?timeout=30" \
      -H "Authorization: Bearer YOUR_API_KEY" --max-time 45
    ```

    </CodeGroup>
  </Accordion>

  <Accordion title="Handle reconnection gracefully">
    If the connection drops, simply resume polling with the last received `chunk_id`. The API is stateless - you won't miss any chunks.
  </Accordion>

  <Accordion title="Use fast mode for lower latency">
    Enable `fast=true` when creating the job for faster rendering, reducing the time between chunks.
  </Accordion>
</AccordionGroup>
