Render with `background_mode=transparent` to get a main video + mask video for compositing onto any background.

## Workflow

<Steps>
  <Step title="Render with transparent mode" icon="play">
    <CodeGroup>

    ```python Python
    import requests, time
    API_KEY, BASE = "YOUR_API_KEY", "https://apis.viggle.ai"
    headers = {"Authorization": f"Bearer {API_KEY}"}

    job = requests.post(f"{BASE}/api/render", headers=headers, data={
        "character_id": "char_xxx",
        "scene_id": "scene_xxx",
        "background_mode": "transparent",
    }).json()
    print(f"Job ID: {job['job_id']}")
    ```

    ```javascript JavaScript
    const API_KEY = "YOUR_API_KEY", BASE = "https://apis.viggle.ai";
    const headers = { Authorization: `Bearer ${API_KEY}` };

    const form = new URLSearchParams();
    form.append("character_id", "char_xxx");
    form.append("scene_id", "scene_xxx");
    form.append("background_mode", "transparent");

    const job = await fetch(`${BASE}/api/render`, {
      method: "POST", headers, body: form
    }).then(r => r.json());
    console.log(`Job ID: ${job.job_id}`);
    ```


## Complete script

<Accordion title="Full end-to-end script">
  <CodeGroup>

  ```python Python
  import requests, time

  API_KEY, BASE = "YOUR_API_KEY", "https://apis.viggle.ai"
  headers = {"Authorization": f"Bearer {API_KEY}"}

  job = requests.post(f"{BASE}/api/render", headers=headers, data={
      "character_id": "char_xxx", "scene_id": "scene_xxx",
      "background_mode": "transparent",
  }).json()

  while True:
      status = requests.get(f"{BASE}/api/render/{job['job_id']}").json()
      if status["status"] == "complete": break
      elif status["status"] == "failed": raise Exception(status.get("error"))
      time.sleep(5)

  for endpoint, filename in [("download", "character.mp4"), ("download/mask", "character_mask.mp4")]:
      resp = requests.get(f"{BASE}/api/render/{job['job_id']}/{endpoint}", headers=headers)
      open(filename, "wb").write(resp.content)
      print(f"Saved {filename}")
  ```

  ```javascript JavaScript
  const fs = require("fs");

  const API_KEY = "YOUR_API_KEY", BASE = "https://apis.viggle.ai";
  const headers = { Authorization: `Bearer ${API_KEY}` };
  const sleep = (ms) => new Promise(r => setTimeout(r, ms));

  // 1. Start transparent render
  const form = new URLSearchParams();
  form.append("character_id", "char_xxx");
  form.append("scene_id", "scene_xxx");
  form.append("background_mode", "transparent");

  const job = await fetch(`${BASE}/api/render`, {
    method: "POST", headers, body: form
  }).then(r => r.json());

  // 2. Poll until complete
  while (true) {
    const status = await fetch(`${BASE}/api/render/${job.job_id}`).then(r => r.json());
    if (status.status === "complete") break;
    if (status.status === "failed") throw new Error(status.error);
    await sleep(5000);
  }

  // 3. Download video and mask
  for (const { endpoint, filename } of [
    { endpoint: "download", filename: "character.mp4" },
    { endpoint: "download/mask", filename: "character_mask.mp4" },
  ]) {
    const resp = await fetch(`${BASE}/api/render/${job.job_id}/${endpoint}`, { headers });
    fs.writeFileSync(filename, Buffer.from(await resp.arrayBuffer()));
    console.log(`Saved ${filename}`);
  }
  ```

  ```bash cURL
  #!/bin/bash
  # Full compositing workflow (requires jq)
  API_KEY="YOUR_API_KEY"
  BASE="https://apis.viggle.ai"

  # 1. Start transparent render
  JOB_ID=$(curl -s -X POST "$BASE/api/render" \
    -H "Authorization: Bearer $API_KEY" \
    -d "character_id=char_xxx" -d "scene_id=scene_xxx" \
    -d "background_mode=transparent" | jq -r '.job_id')
  echo "Job: $JOB_ID"

  # 2. Poll until complete
  while true; do
    STATUS=$(curl -s "$BASE/api/render/$JOB_ID" | jq -r '.status')
    [ "$STATUS" = "complete" ] && break
    [ "$STATUS" = "failed" ] && echo "Failed!" && exit 1
    sleep 5
  done

  # 3. Download video and mask
  curl -s -L -o character.mp4 "$BASE/api/render/$JOB_ID/download" \
    -H "Authorization: Bearer $API_KEY"
  curl -s -L -o character_mask.mp4 "$BASE/api/render/$JOB_ID/download/mask" \
    -H "Authorization: Bearer $API_KEY"
  echo "Saved character.mp4 and character_mask.mp4"
  ```

## Tips

<Accordion title="Green screen alternative">
  If your editor supports chroma keying, use `background_mode=solid` with `bg_color=0,255,0` instead. Traditional green screen without a separate mask file.
</Accordion>
