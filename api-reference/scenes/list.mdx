---
title: 'List Scenes'
api: 'GET /api/scenes'
description: 'List all scenes for your account'
---

Retrieve a list of all scenes associated with your account.

## Response

<ResponseField name="scenes" type="array">
  Array of scene objects
  
  <Expandable title="Scene object">
    <ResponseField name="id" type="string">
      Unique scene identifier
    </ResponseField>
    <ResponseField name="name" type="string">
      Display name
    </ResponseField>
    <ResponseField name="status" type="string">
      One of: `pending`, `processing`, `ready`, `failed`
    </ResponseField>
    <ResponseField name="duration_seconds" type="number">
      Video duration in seconds
    </ResponseField>
    <ResponseField name="fps" type="integer">
      Frames per second
    </ResponseField>
    <ResponseField name="thumbnail_url" type="string">
      URL to scene thumbnail
    </ResponseField>
    <ResponseField name="credits_used" type="integer">
      Credits consumed
    </ResponseField>
    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="integer">
  Total number of scenes
</ResponseField>

<RequestExample>

```bash cURL
curl "https://api.viggle.ai/api/scenes" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```python Python

response = requests.get(

    headers={"Authorization": "Bearer YOUR_API_KEY"}
)

```

