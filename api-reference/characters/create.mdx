Upload a reference image to create a reusable character. Processing runs in background. **Costs 1 credit.**

<Note>
  Best results with clear, front-facing subjects with good lighting and high resolution.
</Note>

## Example

<CodeGroup>
```python Python
import requests

url = "https://apis.viggle.ai/api/characters/preprocess"
headers = {
    "Authorization": "Bearer YOUR_API_KEY",
}
files = {
    "ref_image": ("character.png", open("character.png", "rb"), "image/png"),
}
data = {
    "name": "My Character",
}

response = requests.post(url, headers=headers, files=files, data=data)
result = response.json()

if result.get("error_code"):
    print(f"Error: {result['error_code']} - {result.get('message')}")
else:
    print(f"Character ID: {result['data']['id']}")
    print(f"Status: {result['data']['status']}")
```

```javascript JavaScript
const form = new FormData();
form.append("ref_image", new Blob([await fetch("character.png").then(r => r.blob())]), "character.png");
form.append("name", "My Character");

const response = await fetch("https://apis.viggle.ai/api/characters/preprocess", {
  method: "POST",
  headers: {
    "Authorization": "Bearer YOUR_API_KEY",
  },
  body: form,
});

const result = await response.json();

if (result.error_code) {
  console.error(`Error: ${result.error_code} - ${result.message}`);
} else {
  console.log(`Character ID: ${result.data.id}`);
  console.log(`Status: ${result.data.status}`);
}
```

```bash cURL
curl -X POST https://apis.viggle.ai/api/characters/preprocess \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -F "ref_image=@character.png" \
  -F "name=My Character"
```
