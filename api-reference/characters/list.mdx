---
title: 'List Characters'
api: 'GET /api/characters'
description: 'List all characters for your account'
---

Retrieve a list of all characters associated with your account.

## Response

<ResponseField name="characters" type="array">
  Array of character objects
  
  <Expandable title="Character object">
    <ResponseField name="id" type="string">
      Unique character identifier
    </ResponseField>
    <ResponseField name="name" type="string">
      Display name
    </ResponseField>
    <ResponseField name="status" type="string">
      One of: `pending`, `processing`, `ready`, `failed`
    </ResponseField>
    <ResponseField name="thumbnail_url" type="string">
      URL to character thumbnail image
    </ResponseField>
    <ResponseField name="credits_used" type="integer">
      Credits consumed for this character
    </ResponseField>
    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="integer">
  Total number of characters
</ResponseField>

<RequestExample>

```bash cURL

```
